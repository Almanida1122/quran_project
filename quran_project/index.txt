function startMic() {
  if (!('webkitSpeechRecognition' in window) && !('SpeechRecognition' in window)) {
    setMicStatus('error', '⚠️ آپ کا براؤزر تلاوت نہیں سن سکتا — Google Chrome استعمال کریں');
    return;
  }
  const SR = window.SpeechRecognition || window.webkitSpeechRecognition;
  recognition = new SR();
  recognition.lang = 'ar-SA';
  recognition.continuous = false;
  recognition.interimResults = true;
  recognition.maxAlternatives = 1;

  recognition.onstart = () => {
    isListening = true;
    document.getElementById('micBtn').classList.add('listening');
    document.getElementById('waveBars').querySelectorAll('.wave-bar').forEach(b => {
      b.style.animationDuration = (.4 + Math.random()*.5) + 's';
    });
    setMicStatus('listening', '👂 سن رہا ہوں — آیت پڑھیں...');
    hideBubble();
  };

  recognition.onresult = e => {
    let transcript = '';
    for (let i = e.resultIndex; i < e.results.length; i++)
      transcript += e.results[i][0].transcript;

    const tb = document.getElementById('transcriptBox');
    tb.textContent = transcript;
    tb.classList.add('has-text');

    if (e.results[e.results.length-1].isFinal) processRecitation(transcript);
  };

  recognition.onerror = e => {
    stopMic();
    if (e.error === 'not-allowed')
      setMicStatus('error', '❌ مائیکروفون کی اجازت دیں — براؤزر سیٹنگ چیک کریں');
    else if (e.error === 'no-speech')
      setMicStatus('', '🎤 آواز نہیں آئی — دوبارہ بٹن دبائیں');
    else
      setMicStatus('error', '⚠️ خرابی: ' + e.error + ' — دوبارہ کوشش کریں');
  };

  recognition.onend = () => {
    isListening = false;
    document.getElementById('micBtn').classList.remove('listening');
    if (document.getElementById('micStatus').classList.contains('listening'))
      setMicStatus('', '🎤 مکمل — نتیجہ چیک کریں');
  };

  recognition.start();
}

function stopMic() {
  if (recognition) { try { recognition.stop(); } catch(e){} recognition = null; }
  isListening = false;
  document.getElementById('micBtn').classList.remove('listening');
}

/* ============================================================
   PROCESS RECITATION — word-by-word comparison
============================================================ */
function processRecitation(transcript) {
  if (!currentAyahs.length) return;
  const ayah = currentAyahs[teacherAyahIndex];
  if (!ayah) return;

  const ayahWords   = ayah.text.split(' ');
  const spokenWords = transcript.trim().split(' ');
  const normAyah    = ayahWords.map(normalizeAr);
  const normSpoken  = spokenWords.map(normalizeAr);

  let correct = 0, wrong = 0;
  const results = [];

  normAyah.forEach((w, i) => {
    const spoken = normSpoken[i] || '';
    const ok = spoken === w || similarity(spoken, w) > 0.72;
    results[i] = ok;
    if (ok) correct++; else wrong++;
  });

  // Color the words in this ayah card
  colorWords(teacherAyahIndex, results);

  const pct = Math.round((correct / ayahWords.length) * 100);

  if (pct >= 78) {
    // ✅ Correct
    const card = document.getElementById(`ac-${teacherAyahIndex}`);
    if (card) card.classList.add('recited-correct');
    setMicStatus('', `✅ ماشاءاللہ! ${pct}٪ صحیح`);
    showBubble(randFrom(PRAISE) + `  —  ${pct}٪ صحیح`, false);
    speakText(randFrom(PRAISE));
    recitedCount++;
    updateSurahProgress();
    // Auto-advance after 1.5s
    setTimeout(() => advanceTeacherAyah(), 1500);
  } else {
    // ❌ Mistakes
    setMicStatus('', `❌ ${pct}٪ صحیح — سرخ الفاظ دوبارہ پڑھیں`);
    showBubble(randFrom(WRONG_MSG) + ` (${pct}٪ صحیح)`, true);
    speakText(randFrom(WRONG_MSG));
    if ('vibrate' in navigator) navigator.vibrate([150, 80, 150]);
  }

function advanceTeacherAyah() {
  if (teacherAyahIndex < currentAyahs.length - 1) {
    teacherAyahIndex++;
    highlightTeacherAyah();
  } else {
    setMicStatus('', '🎉 سورہ مکمل! اللہ آپ کو جزا دے۔');
    showBubble('سورہ مکمل! بہت اچھا — ماشاءاللہ', false);
    speakText('ماشاءاللہ، سورہ مکمل ہو گئی');
  }
}

function markCorrect() {
  if (!currentAyahs.length) return;
  const card = document.getElementById(`ac-${teacherAyahIndex}`);
  if (card) card.classList.add('recited-correct');
  recitedCount++;
  updateSurahProgress();
  showToast('✅ آیت صحیح مانی گئی');
  setTimeout(() => advanceTeacherAyah(), 600);
}

function skipAyah() {
  if (!currentAyahs.length) return;

}

  const card = document.getElementById(`ac-${teacherAyahIndex}`);
  if (card) card.classList.remove('recited-correct');

}

/* ============================================================
   SPEECH SYNTHESIS  (TTS Teacher)
============================================================ */
let synth = null;
let voices = [];

function setupSpeechSynthesis() {
  if (!('speechSynthesis' in window)) return;
  synth = window.speechSynthesis;
  const loadVoices = () => { voices = synth.getVoices(); };
  loadVoices();
  synth.onvoiceschanged = loadVoices;
}

function speakText(text) {
  if (!synth) return;
  synth.cancel();
  const utt = new SpeechSynthesisUtterance(text);
  utt.lang = 'ar-SA';
  utt.rate = 0.9;
  utt.pitch = 1.0;
  utt.volume = 0.9;
  // Prefer an Arabic voice if available
  const arVoice = voices.find(v => v.lang.startsWith('ar'));
  if (arVoice) utt.voice = arVoice;
  synth.speak(utt);
}

/* ============================================================
   ARABIC NORMALISATION  (for comparison)
============================================================ */
function normalizeAr(text) {
  return text
    .replace(/[\u064B-\u065F\u0670]/g, '')  // remove harakat (diacritics)
    .replace(/[أإآ]/g, 'ا')
    .replace(/ة/g, 'ه')
    .replace(/ى/g, 'ي')
    .replace(/\s+/g, ' ')
    .trim();
}

/* ============================================================
   LEVENSHTEIN SIMILARITY
============================================================ */
function similarity(a, b) {
  if (a === b) return 1;
  if (!a || !b) return 0;
  const longer  = a.length > b.length ? a : b;
  const shorter = a.length > b.length ? b : a;
  const dist = levenshtein(longer, shorter);
  return (longer.length - dist) / longer.length;
}
function levenshtein(a, b) {
  const dp = Array.from({length: b.length+1}, (_, i) => i);
  for (let i = 1; i <= a.length; i++) {
    let prev = i;
    for (let j = 1; j <= b.length; j++) {
      const val = a[i-1]===b[j-1] ? dp[j-1] : Math.min(dp[j-1], dp[j], prev) + 1;
      dp[j-1] = prev; prev = val;
    }
    dp[b.length] = prev;
  }
  return dp[b.length];
}
